设置定时器时***不用***在onMounted()中执行。

例：
```js
function run() {
    console.log("123");
	// 定时器中执行的逻辑...
}

// 这样就已经触发了定时器
let timer = setInterval(function(){
    run();
}, 2000);

// onMounted(() => {
	// 千万！不要！在这里执行；不需要这么写
    // timer();
// })

onBeforeUnmount(() => {
	// 清除定时器
    clearInterval(timer)
})
```
